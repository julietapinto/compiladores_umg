EZEQUIELAQUIINICIA () [
    // 1. Declaraciones con tipos explícitos y asignación
    int a = 10;
    int b = 5;
    int resultado = 0;
    bool bandera = false;
    
    // Agregamos otros tipos para demostrar que el compilador ya los reconoce
    float pi = 3.14;
    string estado = "Procesando";

    // 2. Lógica (Condicionales y Operadores)
    CHI_LO_HACE (CON a > b && b != 0) [
        resultado = a + b;
        bandera = true;
    ] TONCES [
        resultado = a - b;
        bandera = false;
    ] CHI_NO [
        resultado = 0;
        bandera = false;
    ]

    CHI_LO_HACE (CON !(a < b || b == 0)) [
        resultado = resultado + 100;
        bandera = true;
        estado = "Finalizado";
    ]
    MIENTRAS[CON a > 0] [
        resultado = resultado + a;
        a = a - 1;
    ]
    PARA [ i = 0; CON i < 5; i = i + 1 ] 
    [
        resultado = resultado + i;
    ]
]
EZEQUIELAQUIFINALIZA()
