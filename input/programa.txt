EZEQUIELAQUIINICIA()[
    int a = 5;
    int resultado = 0;
    int i = 0;

    // =========================
    // WHILE
    // =========================
    MIENTRAS[CON a > 0] [
        resultado = resultado + a;
        a = a - 1;
    ]

    // =========================
    // FOR
    // =========================
    PARA[i = 0; CON i < 5; i = i + 1] [
        resultado = resultado + i;
    ]
]
EZEQUIELAQUIFINALIZA()
